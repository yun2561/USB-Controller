uint16_t crc16(const uint8_t*d,uint32_t l,uint16_t c){
    for(uint32_t i=0;i<l;i++){c^=(uint16_t)d[i]<<8;for(int j=0;j<8;j++)c=(c&0x8000)?((c<<1)^0x1021):(c<<1);}return c;
}
uint32_t crc32(const uint8_t*d,uint32_t l,uint32_t c){
    c^=0xFFFFFFFF;for(uint32_t i=0;i<l;i++){c^=d[i];for(int j=0;j<8;j++)c=(c&1)?((c>>1)^0xEDB88320):(c>>1);}return c^0xFFFFFFFF;
}

#include "midi_parser.h"
uint8_t midi_data_len(uint8_t s){
    if(s>=0x80&&s<=0xBF)return 2;if(s>=0xC0&&s<=0xDF)return 1;
    if(s>=0xE0&&s<=0xEF)return 2;
    switch(s){case 0xF1:case 0xF3:return 1;case 0xF2:return 2;default:return 0;}
}
void midi_parser_init(midi_parser_t*p){memset(p,0,sizeof(*p));}
uint8_t midi_parse(midi_parser_t*p,uint8_t b,midi_msg_t*o){
    if(b>=0xF8){o->status=b;o->channel=0;o->data[0]=o->data[1]=0;o->length=1;return 1;}
    if(b==0xF0){p->state=PS_SYSEX;p->msg.status=0xF0;return 0;}
    if(b==0xF7){p->state=PS_IDLE;return 0;}
    if(b&0x80){p->msg.status=b;p->msg.channel=b&0xF;p->running=b;
        uint8_t e=midi_data_len(b);if(!e){*o=p->msg;o->length=1;p->state=PS_IDLE;return 1;}
        p->state=PS_D1;return 0;}
    switch(p->state){
    case PS_IDLE:if(p->running){p->msg.status=p->running;p->msg.channel=p->running&0xF;
        p->msg.data[0]=b;if(midi_data_len(p->running)<=1){*o=p->msg;o->length=2;return 1;}
        p->state=PS_D2;}return 0;
    case PS_D1:p->msg.data[0]=b;if(midi_data_len(p->msg.status)<=1){*o=p->msg;o->length=2;p->state=PS_IDLE;return 1;}
        p->state=PS_D2;return 0;
    case PS_D2:p->msg.data[1]=b;*o=p->msg;o->length=3;p->state=PS_IDLE;return 1;
    default:p->state=PS_IDLE;return 0;}
}
uint8_t midi_encode_msg(const midi_msg_t*m,uint8_t*buf,uint8_t mx){
    if(mx<1)return 0;buf[0]=m->status;uint8_t d=midi_data_len(m->status);
    if(d>=1&&mx>=2)buf[1]=m->data[0];if(d>=2&&mx>=3)buf[2]=m->data[1];return 1+d;
}

uint8_t usb_midi_enc(const midi_msg_t*m,uint8_t cb,usb_midi_pkt_t*p){
    uint8_t cin=(m->status<0xF0)?(m->status>>4):(m->status==0xF6?5:0xF);
    p->cin_cable=((cb&0xF)<<4)|(cin&0xF);p->midi[0]=m->status;p->midi[1]=m->data[0];p->midi[2]=m->data[1];return 1;
}
uint8_t usb_midi_dec(const usb_midi_pkt_t*p,midi_msg_t*m){
    uint8_t cin=p->cin_cable&0xF;m->status=p->midi[0];m->data[0]=p->midi[1];m->data[1]=p->midi[2];
    m->channel=(m->status<0xF0)?m->status&0xF:0;
    switch(cin){case 5:case 0xF:m->length=1;break;case 0xC:case 0xD:case 2:m->length=2;break;default:m->length=3;}
    return 1;
}
uint16_t usb_midi_enc_sysex(const uint8_t*sx,uint16_t len,uint8_t cb,usb_midi_pkt_t*pk,uint16_t mx){
    uint16_t pi=0,i=0;uint8_t cn=(cb&0xF)<<4;
    while(i<len&&pi<mx){uint16_t rem=len-i;
        if(rem>=3){pk[pi].cin_cable=cn|((rem==3&&sx[i+2]==0xF7)?7:4);
            pk[pi].midi[0]=sx[i];pk[pi].midi[1]=sx[i+1];pk[pi].midi[2]=sx[i+2];i+=3;}
        else if(rem==2){pk[pi].cin_cable=cn|6;pk[pi].midi[0]=sx[i];pk[pi].midi[1]=sx[i+1];pk[pi].midi[2]=0;i+=2;}
        else{pk[pi].cin_cable=cn|5;pk[pi].midi[0]=sx[i];pk[pi].midi[1]=0;pk[pi].midi[2]=0;i++;}
        pi++;}return pi;
}

#ifndef MIDI_ROUTER_H
#define MIDI_ROUTER_H

typedef enum{PORT_USB=0,PORT_UART,PORT_INT,PORT_COUNT}port_id_t;
typedef struct{port_id_t id;void(*send)(const uint8_t*,uint16_t);uint16_t(*poll)(uint8_t*,uint16_t);rb_t*tx,*rx;}midi_port_t;
#define MASK_NOTE 1
#define MASK_CC 2
#define MASK_PC 4
#define MASK_PB 8
#define MASK_SX 16
#define MASK_RT 32
#define MASK_ALL 0xFF

#endif

#include "midi_router.h"
#include "debug_rtt.h"
static midi_port_t*s_ports[PORT_COUNT];
static route_rule_t s_rules[PORT_COUNT][PORT_COUNT];
static midi_rx_cb_t s_rxcb=NULL;
static midi_parser_t s_ps[PORT_COUNT];

void router_init(void){
    memset(s_ports,0,sizeof(s_ports));memset(s_rules,0,sizeof(s_rules));
    s_rules[PORT_USB][PORT_UART]=(route_rule_t){1,0xFFFF,MASK_ALL};
    s_rules[PORT_UART][PORT_USB]=(route_rule_t){1,0xFFFF,MASK_ALL};
    for(int i=0;i<PORT_COUNT;i++)midi_parser_init(&s_ps[i]);
}
void router_reg_port(port_id_t id,midi_port_t*p){if(id<PORT_COUNT)s_ports[id]=p;}
void router_set_rule(port_id_t s,port_id_t d,route_rule_t*r){if(s<PORT_COUNT&&d<PORT_COUNT)s_rules[s][d]=*r;}
void router_set_rx_cb(midi_rx_cb_t cb){s_rxcb=cb;}

static uint8_t match(const midi_msg_t*m,const route_rule_t*r){
    if(!r->en)return 0;
    if(m->status<0xF0){if(!(r->ch_mask&(1<<m->channel)))return 0;}
    uint8_t tb=0,s=m->status&0xF0;
    if(s==0x80||s==0x90)tb=MASK_NOTE;else if(s==0xB0)tb=MASK_CC;else if(s==0xC0)tb=MASK_PC;
    else if(s==0xE0)tb=MASK_PB;else if(m->status==0xF0)tb=MASK_SX;else if(m->status>=0xF8)tb=MASK_RT;
    return(r->msg_mask&tb)?1:0;
}
void router_process(port_id_t src,const midi_msg_t*m){
    if(s_rxcb)s_rxcb(src,m);
    for(int d=0;d<PORT_COUNT;d++){if(d==(int)src||!s_ports[d])continue;
        if(match(m,&s_rules[src][d])){uint8_t buf[3];uint8_t l=midi_encode_msg(m,buf,3);
            if(l&&s_ports[d]->send)s_ports[d]->send(buf,l);}}
}
void router_poll_all(void){
    uint8_t buf[64];
    for(int i=0;i<PORT_COUNT;i++){if(!s_ports[i]||!s_ports[i]->poll)continue;
        uint16_t l=s_ports[i]->poll(buf,sizeof(buf));
        for(uint16_t j=0;j<l;j++){midi_msg_t m;if(midi_parse(&s_ps[i],buf[j],&m))router_process((port_id_t)i,&m);}}
}

    case JS_LONGP:if(!raw){c->st=JS_IDLE;e=JE_LONG_REL;}break;

static void upd_indiv(uint32_t now){
    for(int i=0;i<STRIP_LEDS;i++){sl_st_t*l=&s_sl[i];
        if(!l->t0&&l->m!=SL_OFF)l->t0=now;uint32_t el=now-l->t0;
        switch(l->m){
        case SL_OFF:s_out[i]=LED_OFF;break;
        case SL_SOLID:s_out[i]=led_dim(l->c,l->br);break;
        case SL_BLINK:{uint8_t on=l->per?((el/(l->per/2))&1)==0:1;
            s_out[i]=on?led_dim(l->c,l->br):LED_OFF;}break;
        case SL_BREATHE:{if(!l->per){s_out[i]=l->c;break;}
            float t=(float)(el%l->per)/(float)l->per;
            float br=(1.0f-cosf(t*6.2832f))*0.5f;
            s_out[i]=led_dim(l->c,(uint8_t)(br*l->br));}break;
        case SL_PULSE:if(el<l->dur){float f=1.0f-(float)el/(float)l->dur;
            s_out[i]=led_dim(l->c,(uint8_t)(f*l->br));}else{s_out[i]=LED_OFF;l->m=SL_OFF;}break;
        default:break;}
        s_out[i]=led_dim(s_out[i],s_mbr);}
}
static void upd_rainbow(uint32_t now){
    if(!s_rt)s_rt=now;uint32_t el=now-s_rt;uint16_t bh=(uint16_t)((el*360UL/s_rp)%360);
    for(int i=0;i<STRIP_LEDS;i++){uint16_t h=(bh+(uint16_t)(i*360/STRIP_LEDS))%360;
        s_out[i]=led_dim(hsv_to_rgb(h,255,255),s_mbr);}
}
static void upd_vu(void){
    float n=(s_vlv+60)/60;if(n<0)n=0;if(n>1)n=1;uint8_t fill=(uint8_t)(n*STRIP_LEDS);
    float pn=(s_vpk+60)/60;if(pn<0)pn=0;if(pn>1)pn=1;uint8_t pp=(uint8_t)(pn*(STRIP_LEDS-1));
    for(int i=0;i<STRIP_LEDS;i++){
        if(i<fill){float p=(float)i/STRIP_LEDS;led_rgb_t c;
            if(p<0.6f)c=LED_GREEN;else if(p<0.85f)c=(led_rgb_t){255,255,0};else c=LED_RED;
            s_out[i]=led_dim(c,s_mbr);}
        else if(i==pp)s_out[i]=led_dim(LED_WHITE,s_mbr);
        else s_out[i]=LED_OFF;}
}
void strip_update(uint32_t now){
    switch(s_gm){case SL_RAINBOW:upd_rainbow(now);break;case SL_VU:upd_vu();break;
    default:upd_indiv(now);}
    if(!bsp_ws_busy())bsp_ws_update(s_out,STRIP_LEDS);
}

    uint8_t pbuf[FMC_PAGE_SIZE];uint16_t poff;uint32_t faddr;}dfu_ctx_t;

static dfu_prog_cb_t s_dpcb=NULL;static dfu_done_cb_t s_ddcb=NULL;

uint8_t dfu_progress(const dfu_ctx_t*c){return c->total?(uint8_t)(c->rcvd*100/c->total):0;}
static void flush_pg(dfu_ctx_t*c){
    bsp_fmc_erase(c->faddr);bsp_fmc_prog(c->faddr,(const uint32_t*)c->pbuf,c->poff/4);
    c->faddr+=FMC_PAGE_SIZE;c->poff=0;memset(c->pbuf,0xFF,FMC_PAGE_SIZE);
}
void dfu_proc_sysex(dfu_ctx_t*c,const uint8_t*d,uint16_t l){
    if(l<7)return;uint8_t cmd=d[4];
    switch(cmd){
    case DFU_CMD_START:if(l<13)return;
        c->total=((uint32_t)d[5]<<24)|((uint32_t)d[6]<<16)|((uint32_t)d[7]<<8)|d[8];
        c->ecrc=((uint32_t)d[9]<<24)|((uint32_t)d[10]<<16)|((uint32_t)d[11]<<8)|d[12];
        if(c->total>(CONFIG_ADDR-APP_ADDR)){c->st=DFU_ERR;return;}
        c->st=DFU_RX;c->rcvd=0;c->rcrc=0;c->nblk=0;c->poff=0;c->faddr=APP_ADDR;
        memset(c->pbuf,0xFF,FMC_PAGE_SIZE);
        DBG_INFO("DFU","Start sz=%lu",c->total);break;
    case DFU_CMD_DATA:if(c->st!=DFU_RX||l<8)return;
        {uint16_t bn=((uint16_t)d[5]<<8)|d[6];if(bn!=c->nblk){c->st=DFU_ERR;return;}
        uint16_t dl=l-8;const uint8_t*dat=&d[7];
        for(uint16_t i=0;i<dl;i++){c->pbuf[c->poff++]=dat[i];c->rcvd++;
            c->rcrc=crc32(&dat[i],1,c->rcrc);
            if(c->poff>=FMC_PAGE_SIZE)flush_pg(c);}
        c->nblk++;if(s_dpcb)s_dpcb(dfu_progress(c));}break;
    case DFU_CMD_VRFY:if(c->st!=DFU_RX)return;
        if(c->poff){while(c->poff%4)c->pbuf[c->poff++]=0xFF;flush_pg(c);}
        if(c->rcrc==c->ecrc){c->st=DFU_DONE;DBG_INFO("DFU","CRC OK");if(s_ddcb)s_ddcb(1);}
        else{c->st=DFU_ERR;DBG_ERR("DFU","CRC fail");if(s_ddcb)s_ddcb(0);}break;
    case DFU_CMD_BOOT:{uint32_t f=DFU_MAGIC_NORMAL;bsp_fmc_erase(DFU_STAGING_ADDR);
        bsp_fmc_prog(DFU_STAGING_ADDR,&f,1);bsp_system_reset();}break;
    }
}

static ee_rec_t s_cache[CACHE_N];static uint8_t s_cn=0;static uint32_t s_dm=0;
void ea_init(void){s_cn=0;s_dm=0;
    for(uint16_t a=0;a<EE_SIZE&&s_cn<CACHE_N;a+=EE_PGSZ){
        ee_rec_t r;if(bsp_ee_read(a,(uint8_t*)&r,sizeof(r))!=EE_OK)continue;
        if(r.valid==0xAA&&crc16(r.data,28,0xFFFF)==r.crc)s_cache[s_cn++]=r;}
}
ea_st_t ea_read(uint16_t a,uint8_t*d,uint16_t l){
    uint16_t ri=a/28;uint16_t off=a%28;if(ri>=s_cn||off+l>28)return EA_ERR;
    memcpy(d,&s_cache[ri].data[off],l);return EA_OK;
}
ea_st_t ea_write(uint16_t a,const uint8_t*d,uint16_t l){
    uint16_t ri=a/28;uint16_t off=a%28;if(ri>=CACHE_N||off+l>28)return EA_FULL;
    memcpy(&s_cache[ri].data[off],d,l);
    s_cache[ri].crc=crc16(s_cache[ri].data,28,0xFFFF);s_cache[ri].seq++;s_cache[ri].valid=0xAA;
    s_dm|=(1U<<ri);if(ri>=s_cn)s_cn=ri+1;return EA_OK;
}
ea_st_t ea_flush(void){
    for(uint8_t i=0;i<s_cn&&i<32;i++){if(!(s_dm&(1U<<i)))continue;
        uint16_t pa=i*EE_PGSZ*EE_WEAR+(s_cache[i].seq%EE_WEAR)*EE_PGSZ;
        if(pa+EE_PGSZ>EE_SIZE)continue;
        if(bsp_ee_write_page(pa,(const uint8_t*)&s_cache[i],sizeof(ee_rec_t))==EE_OK)s_dm&=~(1U<<i);
        else return EA_BUSY;}return EA_OK;
}
ea_st_t ea_factory_reset(void){
    for(uint16_t i=0;i<CACHE_N;i++){ee_rec_t r;
        uint16_t fa=EEA_FACTORY+i*EE_PGSZ;if(fa+EE_PGSZ>EE_SIZE)break;
        if(bsp_ee_read(fa,(uint8_t*)&r,sizeof(r))==EE_OK&&r.valid==0xAA){
            s_cache[i]=r;s_dm|=(1U<<i);if(i>=s_cn)s_cn=i+1;}}
    return ea_flush();
}
uint8_t ea_dirty(void){return s_dm?1:0;}

#include "gui_engine.h"
#include "bsp_lcd_spi.h"
uint16_t g_strip[LCD_W*LCD_STRIP];
#define MAX_DIRTY 8
static rect_t s_dr[MAX_DIRTY];static uint8_t s_dn=0;static uint8_t s_full=1;
static strip_render_cb_t s_rcb=NULL;
void gui_init(void){s_dn=0;s_full=1;memset(g_strip,0,sizeof(g_strip));}
void gui_invalidate(uint16_t x,uint16_t y,uint16_t w,uint16_t h){
    if(s_dn<MAX_DIRTY)s_dr[s_dn++]=(rect_t){x,y,w,h};else s_full=1;}
void gui_invalidate_all(void){s_full=1;}
void gui_set_render_cb(strip_render_cb_t cb){s_rcb=cb;}
void gui_fill(uint16_t x,uint16_t y,uint16_t w,uint16_t h,uint16_t c){
    for(uint16_t r=y;r<y+h&&r<LCD_STRIP;r++)for(uint16_t col=x;col<x+w&&col<LCD_W;col++)
        g_strip[r*LCD_W+col]=c;}
void gui_flush(void){
    uint8_t sc=LCD_H/LCD_STRIP;
    if(s_full){for(uint8_t s=0;s<sc;s++){memset(g_strip,0,sizeof(g_strip));
            if(s_rcb)s_rcb(s,g_strip,s*LCD_STRIP);
            bsp_lcd_window(0,s*LCD_STRIP,LCD_W-1,(s+1)*LCD_STRIP-1);
            bsp_lcd_data_dma((const uint8_t*)g_strip,LCD_W*LCD_STRIP*2);
            while(bsp_lcd_busy());}
        s_full=0;s_dn=0;
    }else if(s_dn){uint16_t sm=0;
        for(uint8_t i=0;i<s_dn;i++){uint8_t ss=s_dr[i].y/LCD_STRIP,se=(s_dr[i].y+s_dr[i].h-1)/LCD_STRIP;
            for(uint8_t s=ss;s<=se&&s<sc;s++)sm|=(1<<s);}
        for(uint8_t s=0;s<sc;s++){if(!(sm&(1<<s)))continue;
            memset(g_strip,0,sizeof(g_strip));if(s_rcb)s_rcb(s,g_strip,s*LCD_STRIP);
            bsp_lcd_window(0,s*LCD_STRIP,LCD_W-1,(s+1)*LCD_STRIP-1);
            bsp_lcd_data_dma((const uint8_t*)g_strip,LCD_W*LCD_STRIP*2);
            while(bsp_lcd_busy());}
        s_dn=0;}
}
